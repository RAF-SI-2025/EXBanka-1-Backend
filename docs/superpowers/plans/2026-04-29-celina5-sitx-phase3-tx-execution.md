# Celina 5 SI-TX Refactor — Phase 3 TX Execution Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement actual SI-TX TX execution on top of Phase 2's foundation. After this lands, `POST /api/v3/me/transfers` for foreign-prefix receivers performs a real cross-bank transfer through the SI-TX wire protocol; the receiver-side `POST /api/v3/interbank` validates postings, reserves resources, and emits valid `TransactionVote` responses with the 8 SI-TX `NoVote` reasons; sender-side state survives crashes via a replay cron.

**Architecture:** Receiver-side: `PeerTxGRPCHandler` validates incoming `Message<Type>` envelopes via `vote_builder`, executes accepted postings via `posting_executor` (which translates SI-TX postings into existing `account-service` `ReserveIncoming` / `CommitIncoming` / `ReleaseIncoming` RPC calls), and records the response in `peer_idempotence_records` in the same DB transaction as the local commit (per SI-TX spec). Sender-side: `InitiateOutboundTx` debits the sender immediately (preserving existing intra-bank semantics), persists an `outbound_peer_txs` row in `pending`, POSTs `Message<NewTx>` to peer's `/api/v3/interbank` over HTTP, transitions on the response, and is resumable via a 30s `OutboundReplayCron` if the gateway crashes mid-flight. A new internal RPC `ResolvePeerByAPIToken` and `ResolvePeerByBankCode` (full-credentials variants of Phase 2's stubbed admin lookups) close the auth-resolution gap.

**Tech Stack:** Go workspace monorepo, gRPC (`google.golang.org/grpc`), GORM with `glebarez/sqlite` for unit tests, `net/http` for outbound peer calls, existing `account-service` ReserveIncoming/CommitIncoming/ReleaseIncoming RPCs (kept from Phase 1 as SI-TX-shape-compatible), `bcrypt` for token compare, `crypto/hmac` for outbound HMAC signing.

**End state of Phase 3:**
- `POST /api/v3/interbank` with `NEW_TX` → 200 `{type: "YES", transaction_id: "..."}` (or `{type: "NO", noVotes: [...]}` with the 8 SI-TX reasons) instead of 501.
- `POST /api/v3/interbank` with `COMMIT_TX` / `ROLLBACK_TX` → 204.
- `POST /api/v3/me/transfers` with foreign-prefix receiver → 202 with poll URL, real cross-bank transfer initiated.
- `OutboundReplayCron` resumes pending outbound TXs after sender crashes.
- Both auth paths (`X-Api-Key` and HMAC bundle) actually authenticate end-to-end via the new internal RPC.
- `make build` / `make test` / `make lint` all green.

```go
func (h *PeerBankAdminGRPCHandler) ResolvePeerByAPIToken(ctx context.Context, req *transactionpb.ResolvePeerByAPITokenRequest) (*transactionpb.ResolvePeerByAPITokenResponse, error) {
	tok := req.GetApiToken()
	if tok == "" {
		return &transactionpb.ResolvePeerByAPITokenResponse{Found: false}, nil
	}
	rows, err := h.repo.List(true) // active only
	if err != nil {
		return nil, status.Errorf(codes.Internal, "list: %v", err)
	}
	for i := range rows {
		// Constant-time comparison against plaintext (every active peer is small,
		// so iterate; bcrypt-compare is too slow for full-table scan on every
		// inbound request — we keep plaintext to enable O(N) lookup).
		if subtleEqualString(rows[i].APITokenPlaintext, tok) {
			return &transactionpb.ResolvePeerByAPITokenResponse{
				PeerBank: peerBankToFullProto(&rows[i]),
				Found:    true,
			}, nil
		}
	}
	return &transactionpb.ResolvePeerByAPITokenResponse{Found: false}, nil
}

func (h *PeerBankAdminGRPCHandler) ResolvePeerByBankCode(ctx context.Context, req *transactionpb.ResolvePeerByBankCodeRequest) (*transactionpb.ResolvePeerByBankCodeResponse, error) {
	row, err := h.repo.GetByBankCode(req.GetBankCode())
	if err != nil {
		if errors.Is(err, gorm.ErrRecordNotFound) {
			return &transactionpb.ResolvePeerByBankCodeResponse{Found: false}, nil
		}
		return nil, status.Errorf(codes.Internal, "get by bank code: %v", err)
	}
	if !row.Active {
		return &transactionpb.ResolvePeerByBankCodeResponse{Found: false}, nil
	}
	return &transactionpb.ResolvePeerByBankCodeResponse{
		PeerBank: peerBankToFullProto(row),
		Found:    true,
	}, nil
}

func peerBankToFullProto(row *model.PeerBank) *transactionpb.PeerBankFull {
	return &transactionpb.PeerBankFull{
		Id:                row.ID,
		BankCode:          row.BankCode,
		RoutingNumber:     row.RoutingNumber,
		BaseUrl:           row.BaseURL,
		ApiTokenPlaintext: row.APITokenPlaintext,
		HmacInboundKey:    row.HMACInboundKey,
		HmacOutboundKey:   row.HMACOutboundKey,
		Active:            row.Active,
	}
}

// subtleEqualString returns true iff a == b in constant time relative to
// the longer of len(a), len(b). Used to avoid timing leaks in API-token
// comparison.
func subtleEqualString(a, b string) bool {
	if len(a) != len(b) {
		return false
	}
	var v byte
	for i := 0; i < len(a); i++ {
		v |= a[i] ^ b[i]
	}
	return v == 0
}
```

(The `subtle` package's `ConstantTimeCompare` does the same thing for `[]byte` — using `subtleEqualString` here to keep the code self-contained. If lint flags it, swap to `subtle.ConstantTimeCompare([]byte(a), []byte(b)) == 1`.)

```go
// PeerBankResolverAdapter wraps PeerBankAdminServiceClient to satisfy the
// middleware.PeerBankResolver interface. Phase 3 uses the internal
// ResolvePeerByAPIToken / ResolvePeerByBankCode RPCs that return the
// full peer-bank record (including HMAC keys) without exposing it

type PeerBankResolverAdapter struct {
	Client transactionpb.PeerBankAdminServiceClient
}

func (a *PeerBankResolverAdapter) ResolveByBankCode(ctx context.Context, code string) (*middleware.PeerBankRecord, bool, error) {
	resp, err := a.Client.ResolvePeerByBankCode(ctx, &transactionpb.ResolvePeerByBankCodeRequest{BankCode: code})
	if err != nil {
		return nil, false, err
	}
	if !resp.GetFound() {
		return nil, false, nil
	}
	return fullToRecord(resp.GetPeerBank()), true, nil
}

	}
	if !resp.GetFound() {
		return nil, false, nil

}
```

```go
// Package sitx contains transaction-service-side helpers for the SI-TX
// peer protocol: vote building, posting execution, outbound HTTP calls.
// Wire types live in `contract/sitx` and are imported under the alias
// `contractsitx` to avoid a name collision.
package sitx

import (
	contractsitx "github.com/exbanka/contract/sitx"
	"github.com/shopspring/decimal"
)

// BuildPrelimVote runs cheap, in-process validation on an inbound NEW_TX:
// rejects empty postings, rejects postings that don't balance per assetId.
// More expensive checks (account existence, asset acceptability, sufficient
// funds) require account-service calls and are executed by posting_executor
// inside the same DB transaction as the resource reservation.
func BuildPrelimVote(postings []contractsitx.Posting) contractsitx.TransactionVote {
	if len(postings) == 0 {
		return contractsitx.TransactionVote{
			Type:    contractsitx.VoteNo,
			NoVotes: []contractsitx.NoVote{{Reason: contractsitx.NoVoteReasonUnbalancedTx}},
		}
	}
	// Sum debits and credits per assetId.
	netByAsset := map[string]decimal.Decimal{}
	for _, p := range postings {
		net := netByAsset[p.AssetID]
		switch p.Direction {
		case contractsitx.DirectionDebit:
			net = net.Sub(p.Amount)
		case contractsitx.DirectionCredit:
			net = net.Add(p.Amount)
		}
		netByAsset[p.AssetID] = net
	}
	for _, n := range netByAsset {
		if !n.IsZero() {
			return contractsitx.TransactionVote{
				Type:    contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{Reason: contractsitx.NoVoteReasonUnbalancedTx}},
			}
		}
	}
	return contractsitx.TransactionVote{Type: contractsitx.VoteYes}
}
```

## Task 5 — `posting_executor` (executes accepted postings)

The executor takes a YES-voted `[]Posting` and walks the receiver-side credit-postings, calling `account-service.ReserveIncoming(reservation_key="<peer_bank_code>:<idem>")` per posting. Sender-side debit-postings (where `routingNumber == OWN_ROUTING`) get a direct `UpdateBalance` call (this is the receiver-side codepath; sender-side debits are handled by `InitiateOutboundTx` separately, but inbound NEW_TX with debit-postings on our routing number still needs to be rejected — for inbound only credit-postings on our routing number are valid).

**Wait** — re-read SI-TX. NEW_TX postings span both banks; each bank executes only the postings that touch its own `routingNumber`. So for an inbound NEW_TX, we filter postings to those with `routingNumber == OWN_ROUTING` and apply them. If the posting is `CREDIT` we `ReserveIncoming`; if `DEBIT` it's a sender-side leg that the *other* bank must execute (we shouldn't see DEBIT-on-our-routing in a normal flow, but the spec doesn't forbid it — treat as `UNACCEPTABLE_ASSET` since we can't authoritatively debit a peer's account).

Simplification for Phase 3: an inbound NEW_TX must have **at least one CREDIT posting on our routingNumber**. Any DEBIT posting on our routingNumber → `UNACCEPTABLE_ASSET` for that posting. The other bank's postings (different routingNumber) are validated for balance only and ignored for execution.

```go
	"context"

	accountpb "github.com/exbanka/contract/accountpb"
	contractsitx "github.com/exbanka/contract/sitx"
```

```go
// AccountClient is the subset of accountpb.AccountServiceClient that
// posting_executor depends on. Decoupled for testability.
type AccountClient interface {
	GetAccountByNumber(ctx context.Context, in *accountpb.GetAccountByNumberRequest, opts ...interface{}) (*accountpb.AccountResponse, error)
	ReserveIncoming(ctx context.Context, in *accountpb.ReserveIncomingRequest, opts ...interface{}) (*accountpb.ReserveIncomingResponse, error)
	CommitIncoming(ctx context.Context, in *accountpb.CommitIncomingRequest, opts ...interface{}) (*accountpb.CommitIncomingResponse, error)
	ReleaseIncoming(ctx context.Context, in *accountpb.ReleaseIncomingRequest, opts ...interface{}) (*accountpb.ReleaseIncomingResponse, error)
}

// ReserveResult is the outcome of executing the reserve phase of a NEW_TX.
// Vote.Type=YES → all our routingNumber credit-postings reserved
// successfully; Vote.Type=NO → at least one posting failed validation, no
// reservations were committed (caller is responsible for cleanup if any
// partial reservations exist — see ReservationKeys).
type ReserveResult struct {
	Vote            contractsitx.TransactionVote
	ReservationKeys []string // populated on YES; one per credit posting on our routing
}

// PostingExecutor walks an accepted NEW_TX's postings and applies the
// receiver-side reservations via account-service. ownRoutingNumber is
// this bank's routing (e.g. 111) — postings with a different routing
// are not executed locally; they're the responsibility of the other bank.
type PostingExecutor struct {
	accountClient    interface{} // accountpb.AccountServiceClient (boxed so test stubs work)
	ownRouting       int64
}

func NewPostingExecutor(client interface{}, ownRouting int64) *PostingExecutor {
	return &PostingExecutor{accountClient: client, ownRouting: ownRouting}
}

func (e *PostingExecutor) Reserve(ctx context.Context, postings []contractsitx.Posting, peerBankCode, locallyGeneratedKey string) ReserveResult {
	res := ReserveResult{}
	keys := []string{}

	for i := range postings {
		p := postings[i]
		if p.RoutingNumber != e.ownRouting {
			continue // not our leg
		}
		if p.Direction != contractsitx.DirectionCredit {
			// DEBIT on our routingNumber from a peer is not allowed —
			// SI-TX expects each bank to execute only its own credits;
			// peer cannot order us to debit one of our accounts.
			res.Vote = contractsitx.TransactionVote{
				Type: contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{
					Reason:  contractsitx.NoVoteReasonUnacceptableAsset,
					Posting: ptr(i),
				}},
			}
			return res
		}
		// Account-existence + asset-acceptability checks.
		acct, err := e.getAccount(ctx, p.AccountID)
		if err != nil || acct == nil {
			res.Vote = contractsitx.TransactionVote{
				Type: contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{
					Reason:  contractsitx.NoVoteReasonNoSuchAccount,
					Posting: ptr(i),
				}},
			}
			return res
		}
		if !acct.Active {
			res.Vote = contractsitx.TransactionVote{
				Type: contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{
					Reason:  contractsitx.NoVoteReasonUnacceptableAsset,
					Posting: ptr(i),
				}},
			}
			return res
		}
		if acct.CurrencyCode != p.AssetID {
			res.Vote = contractsitx.TransactionVote{
				Type: contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{
					Reason:  contractsitx.NoVoteReasonNoSuchAsset,
					Posting: ptr(i),
				}},
			}
			return res
		}
		// Reserve.
		key := peerBankCode + ":" + locallyGeneratedKey
		if _, err := e.reserveIncoming(ctx, p.AccountID, p.Amount.String(), p.AssetID, key); err != nil {
			// account-service balance-insufficient errors propagate through
			// gRPC as Internal — we map these to INSUFFICIENT_ASSET. For
			// credit postings this is unusual (incoming credit doesn't need
			// balance on receiver) but ReserveIncoming may still fail for
			// other reasons; treat as UNACCEPTABLE_ASSET in that case.
			res.Vote = contractsitx.TransactionVote{
				Type: contractsitx.VoteNo,
				NoVotes: []contractsitx.NoVote{{
					Reason:  contractsitx.NoVoteReasonUnacceptableAsset,
					Posting: ptr(i),
				}},
			}
			return res
		}
		keys = append(keys, key)
	}
	res.Vote = contractsitx.TransactionVote{Type: contractsitx.VoteYes}
	res.ReservationKeys = keys
	return res
}

func (e *PostingExecutor) getAccount(ctx context.Context, accountNumber string) (*accountpb.AccountResponse, error) {

		realC := e.accountClient.(accountpb.AccountServiceClient)
```

```go
	ReserveIncoming(ctx context.Context, in *accountpb.ReserveIncomingRequest, opts ...grpc.CallOption) (*accountpb.ReserveIncomingResponse, error)
	CommitIncoming(ctx context.Context, in *accountpb.CommitIncomingRequest, opts ...grpc.CallOption) (*accountpb.CommitIncomingResponse, error)
	ReleaseIncoming(ctx context.Context, in *accountpb.ReleaseIncomingRequest, opts ...grpc.CallOption) (*accountpb.ReleaseIncomingResponse, error)
}

// ReserveResult is the outcome of executing the reserve phase of a NEW_TX.
type ReserveResult struct {
	Vote            contractsitx.TransactionVote
	ReservationKeys []string // populated on YES; one per credit posting on our routing
}

// PostingExecutor walks an accepted NEW_TX's postings and applies the
// receiver-side reservations via account-service. ownRouting is this
// bank's routing number — postings with a different routing are not
// executed locally.

	client     AccountClient
	ownRouting int64
}

func NewPostingExecutor(client AccountClient, ownRouting int64) *PostingExecutor {
	return &PostingExecutor{client: client, ownRouting: ownRouting}
}

// Reserve runs the receive-side reserve phase of a NEW_TX. It walks the
// postings, reserving each credit-posting on our routingNumber via
// account-service.ReserveIncoming. On any per-posting failure it returns
// a NO vote with the matching SI-TX reason and the failing posting index.
func (e *PostingExecutor) Reserve(ctx context.Context, postings []contractsitx.Posting, peerBankCode, locallyGeneratedKey string) ReserveResult {
	keys := []string{}
	for i := range postings {
		p := postings[i]
		if p.RoutingNumber != e.ownRouting {
			continue
		}
		if p.Direction != contractsitx.DirectionCredit {
			return noVote(contractsitx.NoVoteReasonUnacceptableAsset, i)
		}
		acct, err := e.client.GetAccountByNumber(ctx, &accountpb.GetAccountByNumberRequest{AccountNumber: p.AccountID})
		if err != nil || acct == nil {
			return noVote(contractsitx.NoVoteReasonNoSuchAccount, i)
		}
		if !acct.Active {
			return noVote(contractsitx.NoVoteReasonUnacceptableAsset, i)
		}
		if acct.CurrencyCode != p.AssetID {
			return noVote(contractsitx.NoVoteReasonNoSuchAsset, i)
		}
		key := peerBankCode + ":" + locallyGeneratedKey
		if _, err := e.client.ReserveIncoming(ctx, &accountpb.ReserveIncomingRequest{
			AccountNumber:  p.AccountID,
			Amount:         p.Amount.String(),
			Currency:       p.AssetID,
			ReservationKey: key,
			IdempotencyKey: "sitx-reserve-" + key,
		}); err != nil {
			return noVote(contractsitx.NoVoteReasonUnacceptableAsset, i)
		}
		keys = append(keys, key)
	}
	return ReserveResult{
		Vote:            contractsitx.TransactionVote{Type: contractsitx.VoteYes},
		ReservationKeys: keys,
	}
}

func noVote(reason string, postingIdx int) ReserveResult {
	return ReserveResult{
		Vote: contractsitx.TransactionVote{
			Type: contractsitx.VoteNo,
			NoVotes: []contractsitx.NoVote{{Reason: reason, Posting: ptr(postingIdx)}},
		},
	}
}
```

```go
func (h *PeerTxGRPCHandler) HandleNewTx(ctx context.Context, req *transactionpb.SiTxNewTxRequest) (*transactionpb.SiTxVoteResponse, error) {
	idem := req.GetIdempotenceKey().GetLocallyGeneratedKey()
	peerCode := req.GetPeerBankCode()
	if idem == "" || peerCode == "" {
		return nil, status.Error(codes.InvalidArgument, "missing idempotence_key or peer_bank_code")
	}

	// Replay-cache hit?
	if existing, found, err := h.idemRepo.Lookup(peerCode, idem); err == nil && found {
		var cached transactionpb.SiTxVoteResponse
		if err := json.Unmarshal([]byte(existing.ResponsePayloadJSON), &cached); err == nil {
			return &cached, nil
		}
	} else if err != nil {
		return nil, status.Errorf(codes.Internal, "idem lookup: %v", err)
	}

	postings := protoToPostings(req.GetPostings())

	// Cheap balance check first.
	if vote := sitx.BuildPrelimVote(postings); vote.Type == contractsitx.VoteNo {
		return cacheAndReturn(ctx, h.idemRepo, peerCode, idem, "", voteToProto(vote))
	}

	// Execute reservations.
	res := h.executor.Reserve(ctx, postings, peerCode, idem)
	if res.Vote.Type == contractsitx.VoteNo {
		return cacheAndReturn(ctx, h.idemRepo, peerCode, idem, "", voteToProto(res.Vote))
	}

	txID := uuid.NewString()
	resp := &transactionpb.SiTxVoteResponse{Type: contractsitx.VoteYes, TransactionId: txID}
	return cacheAndReturn(ctx, h.idemRepo, peerCode, idem, txID, resp)
}

func cacheAndReturn(ctx context.Context, repo *repository.PeerIdempotenceRepository, peerCode, idem, txID string, resp *transactionpb.SiTxVoteResponse) (*transactionpb.SiTxVoteResponse, error) {
	payload, _ := json.Marshal(resp)
	rec := &model.PeerIdempotenceRecord{
		PeerBankCode:        peerCode,
		LocallyGeneratedKey: idem,
		TransactionID:       txID,
		ResponsePayloadJSON: string(payload),
	}
	if err := repo.Insert(rec); err != nil {
		return nil, status.Errorf(codes.Internal, "idem insert: %v", err)
	}
	return resp, nil
}

func (h *PeerTxGRPCHandler) HandleCommitTx(ctx context.Context, req *transactionpb.SiTxCommitRequest) (*transactionpb.SiTxAckResponse, error) {
	// Look up the idempotence record to get the transaction_id and the
	// reservations made during NEW_TX.
	idem := req.GetIdempotenceKey().GetLocallyGeneratedKey()
	peerCode := req.GetPeerBankCode()
	if idem == "" || peerCode == "" {
		return nil, status.Error(codes.InvalidArgument, "missing idempotence_key or peer_bank_code")
	}
	rec, found, err := h.idemRepo.Lookup(peerCode, idem)
	if err != nil {
		return nil, status.Errorf(codes.Internal, "lookup: %v", err)
	}
	if !found {
		return nil, status.Error(codes.NotFound, "no NEW_TX record for this idempotence key")
	}
	if rec.TransactionID == "" {
		return nil, status.Error(codes.FailedPrecondition, "previous NEW_TX vote was NO; cannot COMMIT")
	}
	// CommitIncoming is idempotent. We don't have the per-posting list at
	// commit time (it was discarded after NEW_TX); the reservation key is
	// "<peerCode>:<idem>". Account-service commits whatever was reserved
	// under that key.
	key := peerCode + ":" + idem
	if _, err := commitIncoming(ctx, h.client, key); err != nil {
		return nil, status.Errorf(codes.Internal, "commit: %v", err)
	}
	return &transactionpb.SiTxAckResponse{}, nil
}

func (h *PeerTxGRPCHandler) HandleRollbackTx(ctx context.Context, req *transactionpb.SiTxRollbackRequest) (*transactionpb.SiTxAckResponse, error) {
	idem := req.GetIdempotenceKey().GetLocallyGeneratedKey()
	peerCode := req.GetPeerBankCode()
	if idem == "" || peerCode == "" {
		return nil, status.Error(codes.InvalidArgument, "missing idempotence_key or peer_bank_code")
	}
	_, found, err := h.idemRepo.Lookup(peerCode, idem)
	if err != nil {
		return nil, status.Errorf(codes.Internal, "lookup: %v", err)
	}
	if !found {
		// No record → nothing to roll back, treat as success (idempotent).
		return &transactionpb.SiTxAckResponse{}, nil
	}
	key := peerCode + ":" + idem
	if _, err := releaseIncoming(ctx, h.client, key); err != nil {
		return nil, status.Errorf(codes.Internal, "release: %v", err)
	}
	return &transactionpb.SiTxAckResponse{}, nil
}

func (h *PeerTxGRPCHandler) InitiateOutboundTx(ctx context.Context, req *transactionpb.SiTxInitiateRequest) (*transactionpb.SiTxInitiateResponse, error) {
	if h.outRepo == nil || h.httpDial == nil || h.peerLookup == nil {
		return nil, status.Error(codes.Unimplemented, "outbound TX initiation requires WithOutbound; wired in cmd/main.go")
	}
	// Detect peer routing from the to_account_number's 3-digit prefix.
	if len(req.GetToAccountNumber()) < 3 {
		return nil, status.Error(codes.InvalidArgument, "to_account_number too short")
	}
	peerCode := req.GetToAccountNumber()[:3]
	peer, err := h.peerLookup(ctx, peerCode)
	if err != nil || peer == nil {
		return nil, status.Errorf(codes.NotFound, "peer bank %s not registered", peerCode)
	}

	idem := uuid.NewString()
	amt, err := decimal.NewFromString(req.GetAmount())
	if err != nil {
		return nil, status.Errorf(codes.InvalidArgument, "amount: %v", err)
	}
	postings := []contractsitx.Posting{
		{RoutingNumber: peer.OwnRouting, AccountID: req.GetFromAccountNumber(), AssetID: req.GetCurrency(), Amount: amt, Direction: contractsitx.DirectionDebit},
		{RoutingNumber: peer.RoutingNumber, AccountID: req.GetToAccountNumber(), AssetID: req.GetCurrency(), Amount: amt, Direction: contractsitx.DirectionCredit},
	}
	postingsJSON, _ := json.Marshal(postings)
	row := &model.OutboundPeerTx{
		IdempotenceKey: idem,
		PeerBankCode:   peerCode,
		TxKind:         "transfer",
		PostingsJSON:   string(postingsJSON),
		Status:         "pending",
	}
	if err := h.outRepo.Create(row); err != nil {
		return nil, status.Errorf(codes.Internal, "outbound row: %v", err)
	}

	// Sender-side debit-immediate: take the money from the sender now;
	// credit it back on rollback.
	debitKey := fmt.Sprintf("peer-out-debit-%s", idem)
	if _, err := updateBalance(ctx, h.client, req.GetFromAccountNumber(), "-"+req.GetAmount(), debitKey, "interbank debit (Phase 3)"); err != nil {
		return nil, status.Errorf(codes.Internal, "debit: %v", err)
	}

	if err := h.dispatchNewTx(ctx, peer, idem, postings); err != nil {
		// Network error → leave row pending; cron will retry. Don't return
		// error; user gets 202 with poll URL.
	}

	return &transactionpb.SiTxInitiateResponse{
		TransactionId: idem,
		PollUrl:       "/api/v3/me/transfers/" + idem,
		Status:        "pending",
	}, nil
}

// dispatchNewTx is a method on the handler (rather than a free function)
// so subagents can refactor it into the cron loop in Task 9 without
// touching call sites.
func (h *PeerTxGRPCHandler) dispatchNewTx(ctx context.Context, peer *sitx.PeerHTTPTarget, idem string, postings []contractsitx.Posting) error {
	envelope := contractsitx.Message[contractsitx.Transaction]{
		IdempotenceKey: contractsitx.IdempotenceKey{RoutingNumber: peer.OwnRouting, LocallyGeneratedKey: idem},
		MessageType:    contractsitx.MessageTypeNewTx,
		Message:        contractsitx.Transaction{Postings: postings},
	}
	resp, err := h.httpDial.Post(ctx, peer, "/interbank", envelope)
	if err != nil {
		// Mark attempt; cron picks it up.
		_ = h.outRepo.MarkAttempt(idem, err.Error())
		return err
	}
	if resp.Vote.Type == contractsitx.VoteYes {
		// Send COMMIT_TX synchronously.
		commitEnvelope := contractsitx.Message[contractsitx.CommitTransaction]{
			IdempotenceKey: contractsitx.IdempotenceKey{RoutingNumber: peer.OwnRouting, LocallyGeneratedKey: idem},
			MessageType:    contractsitx.MessageTypeCommitTx,
			Message:        contractsitx.CommitTransaction{TransactionID: resp.Vote.TransactionID},
		}
		if _, err := h.httpDial.Post(ctx, peer, "/interbank", commitEnvelope); err != nil {
			_ = h.outRepo.MarkAttempt(idem, err.Error())
			return err
		}
		_ = h.outRepo.MarkCommitted(idem)
	} else {
		_ = h.outRepo.MarkRolledBack(idem, "peer voted NO")
	}
	return nil
}

func protoToPostings(in []*transactionpb.SiTxPosting) []contractsitx.Posting {
	out := make([]contractsitx.Posting, len(in))
	for i, p := range in {
		amt, _ := decimal.NewFromString(p.GetAmount())
		out[i] = contractsitx.Posting{
			RoutingNumber: p.GetRoutingNumber(),
			AccountID:     p.GetAccountId(),
			AssetID:       p.GetAssetId(),
			Amount:        amt,
			Direction:     p.GetDirection(),
		}
	}
	return out
}

func voteToProto(v contractsitx.TransactionVote) *transactionpb.SiTxVoteResponse {
	out := &transactionpb.SiTxVoteResponse{Type: v.Type}
	for _, nv := range v.NoVotes {
		entry := &transactionpb.SiTxNoVote{Reason: nv.Reason}
		if nv.Posting != nil {
			entry.PostingIndex = int32(*nv.Posting)
			entry.PostingIndexSet = true
		}
		out.NoVotes = append(out.NoVotes, entry)
	}
	return out
}
```

```go
package handler

import (
	"bytes"
	"encoding/json"
	"io"
	"net/http"

	"github.com/gin-gonic/gin"

	transactionpb "github.com/exbanka/contract/transactionpb"

// PeerTxDispatcherHandler serves /api/v3/me/transfers and detects whether
// the receiver account number's 3-digit prefix is the bank's own prefix
// (intra-bank → delegates to TransactionHandler.CreateTransfer) or a
// foreign prefix (inter-bank → dispatches to PeerTxService.InitiateOutboundTx
// via gRPC, returning 202 Accepted with a poll URL).
type PeerTxDispatcherHandler struct {
	tx          *TransactionHandler
	peerTx      transactionpb.PeerTxServiceClient
	ownBankCode string
}

func NewPeerTxDispatcherHandler(tx *TransactionHandler, peerTx transactionpb.PeerTxServiceClient, ownBankCode string) *PeerTxDispatcherHandler {
	return &PeerTxDispatcherHandler{tx: tx, peerTx: peerTx, ownBankCode: ownBankCode}
}

func (h *PeerTxDispatcherHandler) CreateTransfer(c *gin.Context) {
	body, err := io.ReadAll(c.Request.Body)
	if err != nil {
		apiError(c, http.StatusBadRequest, ErrValidation, "unreadable body")
		return
	}
	c.Request.Body = io.NopCloser(bytes.NewBuffer(body))

	var req struct {
		FromAccountNumber string `json:"from_account_number"`
		ToAccountNumber   string `json:"to_account_number"`
		ReceiverAccount   string `json:"receiverAccount,omitempty"`
		Amount            string `json:"amount"`
		Currency          string `json:"currency,omitempty"`
	}
	if err := json.Unmarshal(body, &req); err != nil {
		apiError(c, http.StatusBadRequest, ErrValidation, "invalid body")
		return
	}
	receiver := req.ToAccountNumber
	if receiver == "" {
		receiver = req.ReceiverAccount
	}

	if len(receiver) >= 3 && receiver[:3] != h.ownBankCode {
		// Inter-bank dispatch.
		resp, err := h.peerTx.InitiateOutboundTx(c.Request.Context(), &transactionpb.SiTxInitiateRequest{
			FromAccountNumber: req.FromAccountNumber,
			ToAccountNumber:   receiver,
			Amount:            req.Amount,
			Currency:          req.Currency,
		})
		if err != nil {
			handleGRPCError(c, err)
			return
		}
		c.JSON(http.StatusAccepted, gin.H{
			"transaction_id": resp.GetTransactionId(),
			"poll_url":       resp.GetPollUrl(),
			"status":         resp.GetStatus(),
		})
		return
	}
	// Intra-bank.
	h.tx.CreateTransfer(c)
}

func (h *PeerTxDispatcherHandler) GetTransferByID(c *gin.Context) {
	h.tx.GetMyTransfer(c)
}
```
